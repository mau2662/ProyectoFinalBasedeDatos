CREATE TABLE CLIENTE 
(
  ID_CLIENTE NUMBER NOT NULL 
, NOMBRE VARCHAR2(20 BYTE) NOT NULL 
, APELLIDO VARCHAR2(20 BYTE) NOT NULL 
, GENERO VARCHAR2(20 BYTE) 
, CORREO VARCHAR2(50 BYTE) NOT NULL 
, DIRECCION VARCHAR2(100 BYTE) 
, TELEFONO NUMBER NOT NULL 
, CONSTRAINT CLIENTE_PK PRIMARY KEY 
  (
    ID_CLIENTE 
  )
)


CREATE TABLE MEDICAMENTO 
(
  ID_MEDICAMENTO NUMBER NOT NULL 
, NOMBRE VARCHAR2(20 BYTE) NOT NULL 
, FARMACEUTICA VARCHAR2(20 BYTE) 
, FECHA_INGRESO DATE NOT NULL 
, FECHA_VENCIMIENTO DATE NOT NULL 
, CONSTRAINT MEDICAMENTO_PK PRIMARY KEY 
  (
    ID_MEDICAMENTO 
  )
)

CREATE TABLE EMPLEADO 
(
  ID_EMPLEADO NUMBER NOT NULL 
, NOMBRE VARCHAR2(20 BYTE) NOT NULL 
, APELLIDO VARCHAR2(20 BYTE) NOT NULL 
, CORREO VARCHAR2(50 BYTE) NOT NULL 
, TELEFONO NUMBER NOT NULL 
, FECHA_CONTRATACION DATE NOT NULL 
, SALARIO NUMBER NOT NULL 
, CONSTRAINT EMPLEADO_PK PRIMARY KEY 
  (
    ID_EMPLEADO 
  )
)

CREATE TABLE CLIENTE_MEDICAMENTO 
(
  ID_CLIENTE_MEDICAMENTO NUMBER NOT NULL 
, ID_CLIENTE NUMBER NOT NULL 
, ID_MEDICAMENTO NUMBER NOT NULL 
, CONSTRAINT CLIENTE_MEDICAMENTO_PK PRIMARY KEY 
  (
    ID_CLIENTE_MEDICAMENTO 
  )
ADD CONSTRAINT CLIENTE_MEDICAMENTO_FK1      FOREIGN KEY
  (
    ID_CLIENTE 
  )
  REFERENCES CLIENTE
  (
    ID_CLIENTE 
  )
ADD CONSTRAINT CLIENTE_MEDICAMENTO_FK2 FOREIGN KEY
 (
   ID_MEDICAMENTO 
 )
REFERENCES MEDICAMENTO
 (
   ID_MEDICAMENTO 
 )
)
